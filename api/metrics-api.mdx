## Overview

The metrics API provides comprehensive performance data for a specific blockchain chain, including provider statistics, RPC method performance, and system resource usage.

## Endpoint

```
GET /api/metrics/:chain
```

### Path Parameters

<ParamField path="chain" type="string" required>
  The chain identifier (e.g., `ethereum`, `polygon`, `arbitrum`)
</ParamField>

## Response

<ResponseField name="chain" type="string" required>
  The chain identifier
</ResponseField>

<ResponseField name="chain_id" type="integer" required>
  The numeric chain ID (e.g., 1 for Ethereum)
</ResponseField>

<ResponseField name="timestamp" type="integer" required>
  Unix timestamp in milliseconds when metrics were collected
</ResponseField>

<ResponseField name="system_metrics" type="object" required>
  VM and system resource utilization
  
  <Expandable title="system_metrics properties">
    <ResponseField name="memory_mb" type="number">
      Total memory usage in megabytes
    </ResponseField>
    
    <ResponseField name="cpu_percent" type="number">
      CPU utilization percentage
    </ResponseField>
    
    <ResponseField name="process_count" type="integer">
      Number of active Erlang processes
    </ResponseField>
    
    <ResponseField name="run_queue" type="integer">
      Number of processes waiting to run
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="chain_performance" type="object" required>
  Aggregated performance metrics for the chain
  
  <Expandable title="chain_performance properties">
    <ResponseField name="total_calls" type="integer">
      Total number of RPC calls processed
    </ResponseField>
    
    <ResponseField name="success_rate" type="number">
      Success rate percentage (0-100)
    </ResponseField>
    
    <ResponseField name="p50_latency" type="number">
      50th percentile latency in milliseconds
    </ResponseField>
    
    <ResponseField name="p95_latency" type="number">
      95th percentile latency in milliseconds
    </ResponseField>
    
    <ResponseField name="failovers_last_minute" type="integer">
      Number of failover events in the last minute
    </ResponseField>
    
    <ResponseField name="connected_providers" type="integer">
      Number of providers currently connected and healthy
    </ResponseField>
    
    <ResponseField name="total_providers" type="integer">
      Total number of configured providers for this chain
    </ResponseField>
    
    <ResponseField name="recent_activity" type="integer">
      Number of recent RPC calls in the activity buffer
    </ResponseField>
    
    <ResponseField name="rpc_calls_per_second" type="number">
      Current RPC request rate
    </ResponseField>
    
    <ResponseField name="error_rate_percent" type="number">
      Error rate percentage (0-100)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="providers" type="array" required>
  Performance metrics for each provider
  
  <Expandable title="provider properties">
    <ResponseField name="id" type="string">
      Provider identifier
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Human-readable provider name
    </ResponseField>
    
    <ResponseField name="score" type="number">
      Provider performance score (0-100)
    </ResponseField>
    
    <ResponseField name="success_rate" type="number">
      Success rate percentage for this provider
    </ResponseField>
    
    <ResponseField name="total_calls" type="integer">
      Total calls routed to this provider
    </ResponseField>
    
    <ResponseField name="avg_latency_ms" type="number">
      Average response latency in milliseconds
    </ResponseField>
    
    <ResponseField name="calls_last_minute" type="integer">
      Number of calls in the last minute
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="rpc_methods" type="array" required>
  List of RPC methods that have been called on this chain
</ResponseField>

<ResponseField name="rpc_performance_by_provider" type="array">
  Detailed RPC method performance organized by provider
  
  <Expandable title="structure">
    <ResponseField name="provider_id" type="string">
      Provider identifier
    </ResponseField>
    
    <ResponseField name="provider_name" type="string">
      Provider name
    </ResponseField>
    
    <ResponseField name="methods" type="array">
      Performance data for each RPC method on this provider
      
      <Expandable title="method properties">
        <ResponseField name="method" type="string">
          RPC method name (e.g., `eth_blockNumber`)
        </ResponseField>
        
        <ResponseField name="avg_latency_ms" type="number">
          Average latency in milliseconds
        </ResponseField>
        
        <ResponseField name="p50_latency_ms" type="number">
          50th percentile latency
        </ResponseField>
        
        <ResponseField name="p90_latency_ms" type="number">
          90th percentile latency
        </ResponseField>
        
        <ResponseField name="p95_latency_ms" type="number">
          95th percentile latency
        </ResponseField>
        
        <ResponseField name="p99_latency_ms" type="number">
          99th percentile latency
        </ResponseField>
        
        <ResponseField name="success_rate" type="number">
          Success rate for this method (0-1)
        </ResponseField>
        
        <ResponseField name="total_calls" type="integer">
          Number of calls to this method
        </ResponseField>
        
        <ResponseField name="last_updated" type="integer">
          Timestamp of last update
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="rpc_performance_by_method" type="array">
  Detailed RPC method performance organized by method
  
  <Expandable title="structure">
    <ResponseField name="method" type="string">
      RPC method name
    </ResponseField>
    
    <ResponseField name="providers" type="array">
      Performance data for each provider supporting this method (same structure as methods above, with added `provider_id` and `provider_name`)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="last_updated" type="integer" required>
  Timestamp when metrics were last updated
</ResponseField>

## Error Responses

### Chain Not Found

```json
{
  "error": "Chain not found",
  "chain": "invalid-chain",
  "available_chains": ["ethereum", "polygon", "arbitrum"]
}
```

**HTTP Status**: `404 Not Found`

### Missing Chain Parameter

```json
{
  "error": "Chain parameter required",
  "usage": "/api/metrics/{chain_name}",
  "available_chains": ["ethereum", "polygon", "arbitrum"]
}
```

**HTTP Status**: `400 Bad Request`

## Use Cases

### Monitor Chain Performance

Track overall performance metrics for a blockchain:

```bash
curl http://localhost:4000/api/metrics/ethereum
```

### Provider Performance Comparison

Compare performance across different RPC providers:

```bash
# Get provider rankings
curl -s http://localhost:4000/api/metrics/ethereum | \
  jq '.providers | sort_by(-.score) | .[] | {name, score, success_rate, avg_latency_ms}'
```

### RPC Method Analysis

Analyze performance of specific RPC methods:

```bash
# Check eth_getLogs performance across providers
curl -s http://localhost:4000/api/metrics/ethereum | \
  jq '.rpc_performance_by_method[] | select(.method=="eth_getLogs")'
```

### System Resource Monitoring

Monitor system resource usage:

```bash
# Alert if memory usage is high
memory_mb=$(curl -s http://localhost:4000/api/metrics/ethereum | jq '.system_metrics.memory_mb')

if (( $(echo "$memory_mb > 1000" | bc -l) )); then
  echo "Warning: High memory usage: ${memory_mb}MB"
fi
```

### Performance Dashboards

Integrate with monitoring tools like Grafana, Datadog, or Prometheus:

```javascript
// Fetch metrics periodically
setInterval(async () => {
  const response = await fetch('http://localhost:4000/api/metrics/ethereum');
  const metrics = await response.json();
  
  // Send to monitoring backend
  await sendToDatadog({
    'lasso.chain.success_rate': metrics.chain_performance.success_rate,
    'lasso.chain.p95_latency': metrics.chain_performance.p95_latency,
    'lasso.chain.rps': metrics.chain_performance.rpc_calls_per_second,
    'lasso.system.memory_mb': metrics.system_metrics.memory_mb,
    'lasso.system.cpu_percent': metrics.system_metrics.cpu_percent
  });
}, 30000); // Every 30 seconds
```

## Notes

- Metrics are updated in real-time based on actual RPC traffic
- Performance percentiles (p50, p95, etc.) are calculated over a sliding window
- Provider scores are calculated based on a combination of latency, success rate, and reliability
- The `rpc_performance_by_provider` and `rpc_performance_by_method` fields provide the same data organized differently for convenience
- System metrics reflect the Erlang VM's resource usage, not the underlying host system
